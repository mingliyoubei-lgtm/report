def sf(rn, sz=9.5, bd=False, clr=None):
    if clr is None: clr = RGBColor(0x33,0x33,0x33)
    rn.font.name = "Calibri"; rn.font.size = Pt(sz); rn.font.bold = bd; rn.font.color.rgb = clr
    rpr = rn._element.get_or_add_rPr()
    rf = rpr.find(qn("w:rFonts"))
    if rf is None:
        rf = parse_xml('<w:rFonts %s />' % nsdecls("w"))
        rpr.insert(0, rf)

def clr_cell(c):
    p = c.paragraphs[0]
    for r in list(p.runs): r._element.getparent().remove(r._element)
    for ch in list(p._element): p._element.remove(ch)
def mkp(c, txt, sz=8.5, bd=False, clr=None, sb=0, sa=1.5, al=0):
    if clr is None: clr = RGBColor(0x33,0x33,0x33)
    p = c.add_paragraph(); p.alignment = al
    p.paragraph_format.space_before = Pt(sb); p.paragraph_format.space_after = Pt(sa)
    p.paragraph_format.line_spacing = 1.08
    r = p.add_run(txt); sf(r, sz=sz, bd=bd, clr=clr); return p
